        function toggleBreathing() {

            if (!breathingActive) {

                startBreathing();

            }

        }


        function startBreathing() {

            const circle = document.getElementById('breathingCircle');

            const text = document.getElementById('breathingText');

            

            if (!breathingActive) {

                breathingActive = true;

                let phase = 'inhale';

                let seconds = 0;

                

                circle.textContent = 'Breathe In';

                

                breathingInterval = setInterval(() => {

                    seconds++;

                    if (seconds % 30 === 0) {

                        mindfulMinutes++;

                        document.getElementById('mindfulMinutes').textContent = mindfulMinutes;

                    }

                    

                    if (phase === 'inhale') {

                        circle.className = 'breathing-circle inhale';

                        text.textContent = 'Breathe In... 🌬️';

                        phase = 'exhale';

                        attentionScore += 1;

                        document.getElementById('attentionScore').textContent = attentionScore;

                    } else {

                        circle.className = 'breathing-circle exhale';

                        text.textContent = 'Breathe Out... 💨';

                        phase = 'inhale';

                    }

                }, 3000);

            }

        }


        function stopBreathing() {

            if (breathingActive) {

                clearInterval(breathingInterval);

                breathingActive = false;

                const circle = document.getElementById('breathingCircle');

                const text = document.getElementById('breathingText');

                circle.className = 'breathing-circle';

                circle.textContent = 'Great Job!';

                text.textContent = 'Excellent breathing practice! Click to try again.';

            }

        }
